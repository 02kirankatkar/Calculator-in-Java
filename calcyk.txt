    private void pointActionPerformed(java.awt.event.ActionEvent evt) {                                      
      Output_screen.setText(Output_screen.getText()+".");
      
      String str=Output_screen.getText();
      char ch=' ';
      int count=0;
      for(int i=0;i<=str.length()-1;i++)
      {
          ch=str.charAt(i);
          
          if(ch=='.')
          {
              count++;
          }
      }
      if(count>=2)
      {
          JOptionPane.showMessageDialog(this,"Error please Rennter the Digit.","Number Format Error.",JOptionPane.ERROR_MESSAGE);
          Output_screen.setText("0");
      }
    }

    private void equalActionPerformed(java.awt.event.ActionEvent evt) {                                      

        String str=Output_screen.getText();
       
        char ch=' ';
        int cout=0;
        for(int i=0;i<=Output_screen.getText().length()-1;i++)
        {
            ch=str.charAt(i);
            if(ch=='.')
            {
                cout++;
            }
        }
        
        if(cout>=2)
        {
            JOptionPane.showMessageDialog(this,"Error","Number Formate Error.",JOptionPane.ERROR_MESSAGE);
            Output_screen.setText("0");
        }
        else
        {
                if(operator_identifier==1)
       {
           double f=Double.parseDouble(first_number.getText());
           double s=Double.parseDouble(Output_screen.getText());
           double add=f+s;
           
           Output_screen.setText(String.valueOf(add));
       }
       if(operator_identifier==2)
               {
                double f=Double.parseDouble(first_number.getText());
                double s=Double.parseDouble(Output_screen.getText());
                double sub=f-s;

                Output_screen.setText(String.valueOf(sub));
               }
       
        if(operator_identifier==3)
               {
                double f=Double.parseDouble(first_number.getText());
                double s=Double.parseDouble(Output_screen.getText());
                double mul=f*s;

                Output_screen.setText(String.valueOf(mul));
            }
        
        if(operator_identifier==0)
        {
            JOptionPane.showMessageDialog(this,"Enter Second Number","Error",JOptionPane.ERROR_MESSAGE);
            Output_screen.setText("0");
        }
        
          if(operator_identifier==4)
       {
           double f=Double.parseDouble(first_number.getText());
           double s=Double.parseDouble(Output_screen.getText());
           double div=f/s;
           
           Output_screen.setText(String.valueOf(div));
       }
        }
        
    }
